\begin{questions} 
    
 \question Define a subset $Q$ of the xy plane as 
 $Q = \{ (x,y) | 0 \leq y \leq 1-x^2, -1 \leq x \leq 1\}$.

 \begin{parts}
\part Find $\text{area}(Q)$
\part Find $\overline{x} \times  \text{area}(Q)$
\part Find $\overline{y} \times  \text{area}(Q)$
\part Find $\text{vol}_{\text{x-axis}} (Q)$
\part Find $\text{vol}_{\text{y-axis}} (Q)$
 \end{parts}

\question Express the region $Q$ from question 1 in the
form 
\begin{equation}
Q = \{(x,y) | f(y) \leq x \leq g(y), a \leq y \leq b \}.
\end{equation}
Using this representation.

\begin{parts}
    \part Find $\text{area}(Q)$
    \part Find $\overline{y} \times  \text{area}(Q)$
    \part Find $\overline{x} \times   \text{area}(Q)$
    \part Find $\text{vol}_{\text{y-axis}} (Q)$
    \part Find $\text{vol}_{\text{x-axis}} (Q)$
     \end{parts}
 

\end{questions}
